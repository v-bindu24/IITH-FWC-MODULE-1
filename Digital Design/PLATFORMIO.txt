// Define pin numbers for the push buttons
const int buttonW = 2;
const int buttonX = 3;
const int buttonY = 4;
const int buttonZ = 5;

// Define pin numbers for the LEDs
const int ledA = 6;
const int ledB = 7;

// Define boolean variables for inputs
bool W, X, Y, Z;

// Define boolean variables for results
bool A, B;

void setup() {
  // Serial communication for output
  Serial.begin(9600);

  // Set the button pins as inputs with pull-up resistors
  pinMode(buttonW, INPUT_PULLUP);
  pinMode(buttonX, INPUT_PULLUP);
  pinMode(buttonY, INPUT_PULLUP);
  pinMode(buttonZ, INPUT_PULLUP);

  // Set the LED pins as outputs
  pinMode(ledA, OUTPUT);
  pinMode(ledB, OUTPUT);
}

void loop() {
  // Read the state of each push button
  W = digitalRead(buttonW) == LOW;
  X = digitalRead(buttonX) == LOW;
  Y = digitalRead(buttonY) == LOW;
  Z = digitalRead(buttonZ) == LOW;

  // Evaluate logical expressions
  A = (W && X);
  B = (Z && X) || (W && Z) || (Y && Z);

  // Display results on Serial Monitor
  Serial.print("A = ");
  Serial.println(A);

  Serial.print("B = ");
  Serial.println(B);

  // Update LEDs based on results
  digitalWrite(ledA, A ? HIGH : LOW);
  digitalWrite(ledB, B ? HIGH : LOW);

  // Add a small delay to avoid rapid readings
  delay(100);
}
